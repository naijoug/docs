## `lifecycle` - “声明周期”

::: code-tabs

@tab App
```dart
import 'package:flutter/material.dart';

void main() => runApp(MyApp());

class MyApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'Flutter Demo',
      home: Home(),
    );
  }
}

// App 声明周期
class Home extends StatefulWidget {
  @override
  _HomeState createState() => _HomeState();
}

class _HomeState extends State<Home> with WidgetsBindingObserver {
  @override
  void initState() {
    super.initState();
    WidgetsBinding.instance.addObserver(this);  // 注册观察者
  }

  @override
  void dispose() {
    super.dispose();
    WidgetsBinding.instance.removeObserver(this); // 销毁时，移除观察者
  }

  @override
  void didChangeAppLifecycleState(AppLifecycleState state) {
    super.didChangeAppLifecycleState(state);
    if (state == AppLifecycleState.paused) {  // 进入后台
      print('app 进入后台');
    } else if (state == AppLifecycleState.resumed) { // 进入前台
      print('app 进入前台');
    } else if (state == AppLifecycleState.inactive) { // 非激活状态
      print('app 非激活状态');
    } else if (state == AppLifecycleState.suspending) { // 挂起状态
      print('app 挂起状态');
    } 
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text('Flutter App 声明周期')),
      body: Center(
        child: Text('Hello World'),
      ),
    );
  }
}
```

@tab StatefulWidget
```dart
import 'package:flutter/material.dart';

void main() => runApp(MyApp());

class MyApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      home: MyHomePage(),
    );
  }
}

// StatefulWidget 声明周期
class MyHomePage extends StatefulWidget {
  @override
  _MyHomePageState createState() => _MyHomePageState();
}

class _MyHomePageState extends State<MyHomePage>{
  @override
  void initState() {
    super.initState();
    print("initState: 创建状态");
  }
  
  @override
  void didChangeDependencies(){
    super.didChangeDependencies();
    print("didChangeDependencies: 状态改变时调用");
  }

  @override
  Widget build(BuildContext context) {
    print("build：构建 Widget");
    WidgetsBinding.instance.addPostFrameCallback((_) {
        print('addPostFrameCallback: 当前帧的绘制和布局完成后调用');
    });
    return Scaffold(
        appBar: AppBar(
          title: Text("Flutter StatefulWidget 生命周期"),
        ),
        body: Text("生命周期演示"),
    );
  }

  @override
  void didUpdateWidget(covariant MyHomePage oldWidget){
    super.didUpdateWidget(oldWidget);
    print("didUpdateWidget: 当 Widget 中的配置改变时就会调用此函数");
  }

  @override
  void deactivate() {
    super.deactivate();
    print("deactivate：移除时调用");
  }

  @override
  void dispose() {
    super.dispose();
    print("dispose: 销毁时调用");
  }
}
```
