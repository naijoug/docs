                                                                      • 🟢 1137 - 第 N 个泰波那契数open in new window
                                                                      💡

                                                                      【思路】动态规划

                                                                      java
                                                                      class Solution {
    Map<Integer, Integer> dp = new HashMap();
    public int tribonacci(int n) {
        if (n == 0) return 0;
        if (n == 1) return 1;
        if (n == 2) return 1;
        if (dp.containsKey(n)) {
            return dp.get(n);
        }
        int tn = tribonacci(n - 1) + tribonacci(n - 2) + tribonacci(n - 3);
        dp.put(n, tn);
        return tn;
    }
}
                                                                      swift
                                                                      class Solution {
    private var dp: [Int: Int] = [:]
    func tribonacci(_ n: Int) -> Int {
        switch n {
        case 0: return 0
        case 1: return 1
        case 2: return 1
        default:
            if let fn = dp[n] { return fn }
            let fn = tribonacci(n - 3) + tribonacci(n - 2) + tribonacci(n - 1)
            dp[n] = fn
            return fn
        }
    }
}
